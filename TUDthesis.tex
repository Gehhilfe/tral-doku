\newcommand\processColon@json{%
  \colon@json%

    \global\colonfoundonthislinetrue%
  \fi
}

  \ifcolonfoundonthisline%

      \def\lst@thestyle{\JSONnumbervaluestyle}%
    \fi
  \fi
  %override by keyword style if a keyword is detected!
  \lsthk@DetectKeywords% 
}
